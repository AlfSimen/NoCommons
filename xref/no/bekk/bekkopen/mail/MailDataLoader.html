27  	public static void loadFromInputStream(InputStream is) throws IOException {
28  		if (is == null) {
29  			throw new IllegalArgumentException();
30  		}
31  		Map<Poststed, List<Postnummer>> poststedMap = new HashMap<Poststed, List<Postnummer>>();
32  		Map<Postnummer, Poststed> postnummerMap = new HashMap<Postnummer, Poststed>();
33  		InputStreamReader isr = new InputStreamReader(is, Charset.forName("UTF-8"));
34  		BufferedReader br = new BufferedReader(isr);
35  		String line;
36  		while ((line = br.readLine()) != null) {
37  			StringTokenizer st = new StringTokenizer(line, "\t", false);
38  			Postnummer pn = MailValidator.getPostnummer(st.nextToken());
39  			Poststed ps = new Poststed(st.nextToken());
40  
41  			// add to poststedMap
42  			List<Postnummer> postnummerList = new ArrayList<Postnummer>();
43  			if (poststedMap.containsKey(ps)) {
44  				postnummerList = poststedMap.get(ps);
45  			}
46  			if (!postnummerList.contains(pn)) {
47  				postnummerList.add(pn);
48  			}
49  			poststedMap.put(ps, postnummerList);
50  
51  			// add to postnummerMap
52  			if (!postnummerMap.containsKey(pn)) {
53  				postnummerMap.put(pn, ps);
54  			}
55  		}
56  		br.close();
57  		isr.close();
58  		MailValidator.setPoststedMap(poststedMap);
59  		MailValidator.setPostnummerMap(postnummerMap);
60  	}
