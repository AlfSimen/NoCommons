27  	private static List<Kontonummer> getKontonummerListUsingGenerator(KontonummerDigitGenerator generator, int length) {
28  		List<Kontonummer> result = new ArrayList<Kontonummer>();
29  		int numAddedToList = 0;
30  		while (numAddedToList < length) {
31  			Kontonummer kontoNr;
32  			try {
33  				kontoNr = KontonummerValidator.getAndForceValidKontonummer(generator.generateKontonummer());
34  			} catch (IllegalArgumentException iae) {
35  				// this number has no valid checksum
36  				continue;
37  			}
38  			result.add(kontoNr);
39  			numAddedToList++;
40  		}
41  		return result;
42  	}

56  	public static List<Kontonummer> getKontonummerListForAccountType(String accountType, int length) {
57  		KontonummerValidator.validateAccountTypeSyntax(accountType);
58  		final class AccountTypeKontonrDigitGenerator extends KontonummerDigitGenerator {
59  			private final String accountType;
60  
61  			AccountTypeKontonrDigitGenerator(String accountType) {
62  				this.accountType = accountType;
63  			}
64  
65  			@Override
66  			String generateKontonummer() {
67  				StringBuffer kontonrBuffer = new StringBuffer(LENGTH);
68  				for (int i = 0; i < LENGTH;) {
69  					if (i == ACCOUNTTYPE_START_DIGIT) {
70  						kontonrBuffer.append(accountType);
71  						i += accountType.length();
72  					} else {
73  						kontonrBuffer.append((int) (Math.random() * 10));
74  						i++;
75  					}
76  				}
77  				return kontonrBuffer.toString();
78  			}
79  		}
80  		return getKontonummerListUsingGenerator(new AccountTypeKontonrDigitGenerator(accountType), length);
81  	}
