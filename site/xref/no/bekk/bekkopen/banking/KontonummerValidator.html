10  public class KontonummerValidator extends StringNumberValidator {

13  	protected static final int ACCOUNTTYPE_NUM_DIGITS = 2;
14  	protected static final int REGISTERNUMMER_NUM_DIGITS = 4;

20      /***
21       * Return true if the provided String is a valid Kontonummmer.
22       * 
23       * @param kontonummer
24       *            A String containing a Kontonummer
25       * @return true or false
26       */
27      public static boolean isValid(String kontonummer) {
28          boolean result = false;
29          try {
30              KontonummerValidator.getKontonummer(kontonummer);
31              result = true;
32          } catch (IllegalArgumentException e) {
33              // ignore
34          }
35          return result;
36      }

38      /***
39       * Returns an object that represents a Kontonummer.
40       * 
41       * @param kontonummer
42       *            A String containing a Kontonummer
43       * @return A Kontonummer instance
44       * @throws IllegalArgumentException
45       *                thrown if String contains an invalid Kontonummer
46       */
47      public static Kontonummer getKontonummer(String kontonummer) 
48          throws IllegalArgumentException {
49          validateSyntax(kontonummer);
50          validateChecksum(kontonummer);
51          return new Kontonummer(kontonummer);
52      }

54      /***
55       * Returns an object that represents a Kontonummer. The checksum of the supplied kontonummer
56       * is changed to a valid checksum if the original kontonummer has an invalid checksum.
57       * 
58       * @param kontonummer
59       * 			A String containing a Kontonummer
60       * @return A Kontonummer instance
61       * @throws IllegalArgumentException
62   	 * thrown if String contains an invalid Kontonummer, ie. a number which for one cannot
63   	 * 			calculate a valid checksum.
64       */
65      public static Kontonummer getAndForceValidKontonummer(String kontonummer) {
66      	validateSyntax( kontonummer );
67      	try {
68      		validateChecksum( kontonummer );
69      	}
70      	catch( IllegalArgumentException iae )  {
71      		Kontonummer k = new Kontonummer( kontonummer );
72      		int checksum = calculateMod11CheckSum(getMod11Weights(k), k );
73      		kontonummer = kontonummer.substring( 0, LENGTH - 1 ) + checksum;
74      	}
75      	return new Kontonummer( kontonummer );
76      }
77  
78      static void validateSyntax(String kontonummer) {
79          validateLengthAndAllDigits(kontonummer, LENGTH);
80      }
81      
82      static void validateAccountTypeSyntax(String accountType) {
83      	validateLengthAndAllDigits(accountType, ACCOUNTTYPE_NUM_DIGITS);
84      }
85      
86      static void validateRegisternummerSyntax(String registernummer) {
87      	validateLengthAndAllDigits(registernummer, REGISTERNUMMER_NUM_DIGITS);
88      }
89  
90      static void validateChecksum(String kontonummer) {
91          Kontonummer k = new Kontonummer(kontonummer);
92          int k1 = calculateMod11CheckSum(getMod11Weights(k), k);
93          if (k1 != k.getChecksumDigit()) {
94              throw new IllegalArgumentException(ERROR_INVALID_CHECKSUM
95                      + kontonummer);
96          }
97      }
